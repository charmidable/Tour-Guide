package tourGuide.service;

import gpsUtil.location.Attraction;
import gpsUtil.location.VisitedLocation;

public class RewardTuple
{
    private VisitedLocation visitedLocationBean;
    private Attraction attractionBean;

    public RewardTuple(VisitedLocation visitedLocationBean, Attraction attractionBean) {
        this.visitedLocationBean = visitedLocationBean;
        this.attractionBean = attractionBean;
    }

    public VisitedLocation getVisitedLocationBean() {
        return visitedLocationBean;
    }

    public void setVisitedLocationBean(VisitedLocation visitedLocationBean) {
        this.visitedLocationBean = visitedLocationBean;
    }

    public Attraction getAttractionBean() {
        return attractionBean;
    }

    public void setAttractionBean(Attraction attractionBean) {
        this.attractionBean = attractionBean;
    }
}
